```javascript
function calculateROI(template) {
  const investment = {
    buildTime: template.hours * $100, // Your hourly rate
    monthlyAICosts: template.aiCost,
    platformFee: template.price * 0.3
  };

  const revenue = {
    perCustomer: template.price * 0.7, // Your 70% share
    break_even_customers: Math.ceil(investment.buildTime / revenue.perCustomer),
    monthly_at_10_customers: revenue.perCustomer * 10,
    annual_at_50_customers: revenue.perCustomer * 50 * 12
  };

  return {
    breakEven: `${revenue.break_even_customers} customers`,
    monthlyPotential: `$${revenue.monthly_at_10_customers}`,
    annualPotential: `$${revenue.annual_at_50_customers}`
  };
}

// Example: Lead Scoring System

  hours: 20,
  price: 349,
  aiCost: 15
});
```
